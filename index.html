#1C]Code to check whether a positive integer is prime or not prime


num = int(input("Enter a number: "))
if num < 2:
    print("Not Prime")
else:
    for i in range(2, num):
        if num % i == 0:
            print("Not Prime")
            break;
    else:
        print("Prime")

#2A]Code to print compound and simple interest


p = float(input("Enter the principal value: "))
r = float(input("Enter the rate: "))
t = float(input("Enter time provided: "))
SI = (p * r * t) / 100
print(f"Simple interest - {SI}")
CI = p * ((1 + r / 100) ** t - 1)
print(f"Compound interest - {CI}")

#5A]Code to display all prime numbers between 1-100 using functions


def is_prime(num):
    if num <= 1:
       return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def display_prime():
    print("Prime numbers between 1-100")
    for number in range(1, 101):
        if is_prime(number):
            print(number, end=" ")
display_prime()

#9B]Code to demonstrate the concept of method and message passing


class Dog:
    def __init__(self, name):
        self.name = name

    def display_info(self):
        print(f"Name: {self.name}")

    def bark(self):
        print(f"{self.name} says: Woof! Woof!")

my_dog = Dog(input("Enter your dog's name: "))

my_dog.display_info()
my_dog.bark()

#9C]Code to demonstrate the concept of encapsulation and access specifiers


class SecretAgent:
    def __init__(self, name, code_name, mission):
        self.name = name        # public
        self._code_name = code_name  # protected
        self.__mission = mission  # private

    def introduce(self):
        print(f"Agent {self.name} here! My code name is {self._code_name}. Shh, it's a secret!")

    def get_mission(self):
        print(f"Mission: {self.__mission}")

agent = SecretAgent("James", "007", "Save the world")

print(f"Agent {agent.name} reporting in!")

print(f"My code name is {agent._code_name}")

# The next line will throw an error as '__mission' is private
# print(f"My mission is {agent.__mission}")

agent.introduce()
agent.get_mission()

#10A]Code to demonstrate the concept of inheritance and polymorphism


class Animal:
    def __init__(self, name):
        self.name = name

    def make_sound(self):
        pass

class Dog(Animal):
    def make_sound(self):
        print(f"{self.name} says: Woof! Woof!")

class Cat(Animal):
    def make_sound(self):
        print(f"{self.name} says: Meow!")

my_dog = Dog("Coco")
my_cat = Cat("Bob")

my_dog.make_sound()
my_cat.make_sound()
