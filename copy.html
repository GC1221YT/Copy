#5A]Code to display all prime numbers between 1-100 using functions


def is_prime(num):
    if num <= 1:
       return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def display_prime():
    print("Prime numbers between 1-100")
    for number in range(1, 101):
        if is_prime(number):
            print(number, end=" ")
display_prime()

#9C]Code to demonstrate the concept of encapsulation and access specifiers


class SecretAgent:
    def __init__(self, name, code_name, mission):
        self.name = name        # public
        self._code_name = code_name  # protected
        self.__mission = mission  # private

    def introduce(self):
        print(f"Agent {self.name} here! My code name is {self._code_name}. Shh, it's a secret!")

    def get_mission(self):
        print(f"Mission: {self.__mission}")

agent = SecretAgent("James", "007", "Save the world")

print(f"Agent {agent.name} reporting in!")

print(f"My code name is {agent._code_name}")

# The next line will throw an error as '__mission' is private
# print(f"My mission is {agent.__mission}")

agent.introduce()
agent.get_mission()

#10A]Code to demonstrate the concept of inheritance and polymorphism


class Animal:
    def __init__(self, name):
        self.name = name

    def make_sound(self):
        pass

class Dog(Animal):
    def make_sound(self):
        print(f"{self.name} says: Woof! Woof!")

class Cat(Animal):
    def make_sound(self):
        print(f"{self.name} says: Meow!")

my_dog = Dog("Coco")
my_cat = Cat("Bob")

my_dog.make_sound()
my_cat.make_sound()

#8B-Code to reverse a string using recursion

This code demonstrates how to reverse a string using recursion:


def reverse(string):
    if len(string) == 0:
        return
    temp = string[0]
    reverse(string[1:])
    print(temp, end=" ")

string = input("Enter a string: ")
reverse(string)

def student_details():
    num_students = int(input("Enter the number of students: "))
    file_name = "student_records.txt"

    with open(file_name, "w") as file:
        for _ in range(num_students):
            name = input("Enter student's name: ")
            marks = float(input("Enter student's marks: "))
            file.write(f"{name}: {marks}\n")

    print(f"Student details saved successfully in {file_name}")

student_details()
